# Cache zones
proxy_cache_path /var/cache/nginx/claudyne levels=1:2 keys_zone=claudyne_cache:10m max_size=1g inactive=60m use_temp_path=off;

# Upstream definitions avec health checks
upstream claudyne_backend {
    server 127.0.0.1:3001 max_fails=3 fail_timeout=30s;
    keepalive 32;
}

# HTTP to HTTPS redirect
server {
    listen 80;
    listen [::]:80;
    server_name claudyne.com www.claudyne.com;

    # Security headers même en HTTP
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-Frame-Options "DENY" always;

    # Redirect to HTTPS
    return 301 https://claudyne.com$request_uri;
}

# HTTPS Production Server - Grade A+
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name claudyne.com www.claudyne.com;

    # SSL Certificates (Let's Encrypt)
    ssl_certificate /etc/letsencrypt/live/claudyne.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/claudyne.com/privkey.pem;
    ssl_trusted_certificate /etc/letsencrypt/live/claudyne.com/chain.pem;

    # Security Headers - Grade A+
    add_header Strict-Transport-Security "max-age=63072000; includeSubDomains; preload" always;
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    add_header Permissions-Policy "camera=(), microphone=(), geolocation=()" always;

    # CSP optimisé pour Claudyne
    add_header Content-Security-Policy "default-src 'self'; script-src 'self' 'unsafe-inline' 'unsafe-eval' https://cdnjs.cloudflare.com https://cdn.jsdelivr.net; style-src 'self' 'unsafe-inline' https://fonts.googleapis.com https://cdnjs.cloudflare.com; font-src 'self' https://fonts.gstatic.com https://cdnjs.cloudflare.com; img-src 'self' data: https: blob:; connect-src 'self' https://claudyne.com wss://claudyne.com; media-src 'self' blob:; object-src 'none'; base-uri 'self'; form-action 'self';" always;

    # Connection limits
    limit_conn perip 10;
    limit_conn perserver 1000;

    # Logging
    access_log /var/log/nginx/claudyne-access.log main;
    error_log /var/log/nginx/claudyne-error.log warn;

    # Error pages
    error_page 404 /404.html;
    error_page 500 502 503 504 /50x.html;
    error_page 429 /429.html;

    # Health check (sans rate limiting)
    location = /health {
        proxy_pass http://claudyne_backend/health;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        access_log off;

        # Health check timeouts
        proxy_connect_timeout 5s;
        proxy_send_timeout 5s;
        proxy_read_timeout 5s;
    }

    # API Routes avec rate limiting strict
    location ^~ /api/ {
        limit_req zone=api burst=20 nodelay;
        limit_req_status 429;

        proxy_pass http://claudyne_backend;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;

        # API Timeouts optimisés
        proxy_connect_timeout 30s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;

        # Buffer settings pour API
        proxy_buffering on;
        proxy_buffer_size 16k;
        proxy_buffers 8 16k;
        proxy_busy_buffers_size 32k;

        # Pas de cache pour les API
        add_header Cache-Control "no-cache, no-store, must-revalidate" always;
        add_header Pragma "no-cache" always;
        add_header Expires "0" always;
    }

    # Routes d'administration avec sécurité renforcée
    location ^~ /admin {
        limit_req zone=admin burst=10 nodelay;
        limit_req_status 429;

        proxy_pass http://claudyne_backend;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;

        # Headers sécurisés pour admin
        add_header X-Frame-Options "DENY" always;
        add_header Cache-Control "no-cache, no-store, must-revalidate" always;
    }

    # Interface mobile optimisée
    location ^~ /mobile {
        limit_req zone=mobile burst=30 nodelay;
        limit_req_status 429;

        proxy_pass http://claudyne_backend;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;

        # Optimisations pour mobiles (réseaux lents)
        proxy_buffering off;
        proxy_request_buffering off;
        proxy_connect_timeout 15s;
        proxy_send_timeout 30s;
        proxy_read_timeout 30s;

        add_header X-Mobile-Optimized "true" always;
    }

    # WebSocket pour temps réel
    location ^~ /socket.io/ {
        proxy_pass http://claudyne_backend;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # WebSocket timeouts
        proxy_read_timeout 86400s;
        proxy_send_timeout 86400s;
        proxy_connect_timeout 60s;
    }

    # Fichiers statiques avec cache optimal
    location ~* \.(jpg|jpeg|png|gif|ico|svg|webp|avif)$ {
        expires 1y;
        add_header Cache-Control "public, immutable, no-transform" always;
        add_header Vary "Accept-Encoding" always;
        access_log off;

        # Compression pour images SVG
        location ~* \.svg$ {
            add_header Content-Type "image/svg+xml" always;
        }
    }

    # CSS et JavaScript avec versioning
    location ~* \.(css|js|woff|woff2|ttf|eot|otf)$ {
        expires 1y;
        add_header Cache-Control "public, immutable, no-transform" always;
        add_header Vary "Accept-Encoding" always;
        access_log off;
    }

    # Documents et médias
    location ~* \.(pdf|doc|docx|ppt|pptx|xls|xlsx|mp4|mp3|avi|mov)$ {
        expires 30d;
        add_header Cache-Control "public, no-transform" always;
        add_header X-Content-Type-Options "nosniff" always;

        # Headers pour téléchargement sécurisé
        add_header Content-Disposition "attachment" always;
    }

    # Manifest et service workers
    location ~* \.(webmanifest|manifest\.json)$ {
        expires 24h;
        add_header Cache-Control "public, no-transform" always;
        add_header Content-Type "application/manifest+json" always;
    }

    # Robots et sitemap
    location = /robots.txt {
        expires 24h;
        add_header Cache-Control "public, no-transform" always;
        access_log off;
    }

    location = /sitemap.xml {
        expires 24h;
        add_header Cache-Control "public, no-transform" always;
        access_log off;
    }

    # Sécurité : bloquer fichiers sensibles
    location ~ /\. {
        deny all;
        access_log off;
        log_not_found off;
    }

    location ~* \.(env|config|log|sql|bak|backup|old)$ {
        deny all;
        access_log off;
        log_not_found off;
    }

    # Interface principale (catch-all)
    location / {
        limit_req zone=general burst=50 nodelay;
        limit_req_status 429;

        proxy_pass http://claudyne_backend;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;

        # Timeouts standards
        proxy_connect_timeout 30s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;

        # Fallback pour SPA
        proxy_intercept_errors on;
        error_page 404 = @fallback;
    }

    # Fallback pour Single Page Applications
    location @fallback {
        proxy_pass http://claudyne_backend;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}

# Monitoring et status
server {
    listen 127.0.0.1:8080;
    server_name localhost;
    access_log off;

    location /nginx-status {
        stub_status on;
        allow 127.0.0.1;
        deny all;
    }

    location /health-check {
        return 200 "Claudyne Nginx OK\n";
        add_header Content-Type text/plain always;
    }
}
